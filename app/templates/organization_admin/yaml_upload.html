YAML File Format

Your YAML file should follow this structure:

value_types:
  - name: "Cost"
    kind: numeric
    numeric_format: decimal
    decimal_places: 2
    unit_label: "€"
    default_aggregation_formula: sum

spaces:
  - id: S1
    name: "Season 1"
    description: "First season"
    space_label: "Season"
    display_order: 1

    challenges:
      - id: C1
        name: "Digital Transformation"
        description: "Transform our business"
        display_order: 1

        initiatives:
          - id: I1
            name: "Standard processes"
            description: "Implement E2E processes"

            systems:
              - id: SYS1
                name: "ERP System"
                description: "Core ERP"

                kpis:
                  - name: "Cost reduction"
                    description: "Reduce costs by 15%"
                    value_types:
                      - name: "Cost"
                        colors:
                          positive: "#dc3545"  # red
                          zero: "#6c757d"      # gray
                          negative: "#28a745"  # green
